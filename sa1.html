                int push(int arr[],int size,int value,int i){
                   if(i > size-1){
                       printf("Full");
            
                   }
                   else{
                       i++;
                       arr[i]=value;
                   }
                   return i;
                }
                int pop(int arr[],int size,int i){
                    if(i == -1){
                        printf("Stack is empty");
            
                    }else{
                        i--;
                    }
                    return i;
                }
                int isEmpty(int i){
                    if(i == -1){
                        printf("\nStack is empty");
            
                    }
                    else{
                    printf("\nStack is not empty yet");
                    }
                }
                void isFull(int size,int i){
                    if(i > size-1){
                        printf("\nStack is full");
                        return;
                    }
                    printf("\nStack is not full yet");
                }
                void display(int arr[],int size,int i){
                    int k;
                    for(k=0;k<=i;k++){
                        printf("%d ",arr[k]);
                    }
                }
