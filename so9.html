                void merge(int arr[], int brr[], int crr[], int l1, int r1) {
                    int i = 0, j = 0, k = 0;
                
                    while (i < l1 && j < r1) {
                        if (brr[i] <= crr[j]) {
                            arr[k] = brr[i];
                            i++;
                        } else {
                            arr[k] = crr[j];
                            j++;
                        }
                        k++;
                    }
                
                    while (i < l1) {
                        arr[k] = brr[i];
                        i++;
                        k++;
                    }
                
                    while (j < r1) {
                        arr[k] = crr[j];
                        j++;
                        k++;
                    }
                }
                
                void mergeSort(int arr[], int size) {
                    if (size <= 1) {
                        return; // Base case: Arrays of size 1 or less are already sorted.
                    }
                
                    int mid = size / 2;
                    int left[mid];
                    int right[size - mid];
                
                    for (int i = 0; i < mid; i++) {
                        left[i] = arr[i];
                    }
                
                    for (int i = mid; i < size; i++) {
                        right[i - mid] = arr[i];
                    }
                
                    mergeSort(left, mid);
                    mergeSort(right, size - mid);
                
                    merge(arr, left, right, mid, size - mid);
                }
                
                int main() {
                    int size;
                    printf("Enter the size of the array: ");
                    scanf("%d", &size);
                
                    int arr[50];
                
                    printf("Enter the elements:\n");
                    for (int i = 0; i < size; i++) {
                        printf("Element %d: ", i + 1);
                        scanf("%d", &arr[i]);
                    }
                
                    mergeSort(arr, size);
                
                    printf("Sorted elements are: ");
                    for (int i = 0; i < size; i++) {
                        printf("%d ", arr[i]);
                    }
                    printf("\n");
                }
