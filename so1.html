//Selection Sort       
                #include<stdio.h>
                    int selectionSort(int arr[],int size){
                        int i,j;
                        int minIndex;
                        for(i=0;i<size-1;i++){
                            minIndex=i;
                            for(j=i+1;j<size;j++){
                                if(arr[j] < arr[minIndex]){
                                    minIndex=j;
                                }
                            }
                             int temp=arr[i];
                                arr[i]=arr[minIndex];
                                arr[minIndex]=temp;
                        }
                        printf("Sorted array: ");
                        for(i=0;i<size;i++){
                            printf("%d ",arr[i]);
                        }
                         printf("\n");
                    }
                    void main(){
                        int arr[20]={20,4,50,6,9};
                        selectionSort(arr,5);
                        
                    }  
                        
                    
 
                        
OUTPUT

                        Sorted array: 4 6 9 20 50
