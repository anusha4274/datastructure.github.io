// calculate factorial of a number using recursion

int fact(int n)
{
    if(n==0)
    {
        return 1;
    }
    else
    {
        return n*fact(n-1);
    }
}
void main()
{
    unsigned int n;
    printf("enter a positive number ");
    scanf("%u",&n);
    printf("factorial is %dÂ ",fact(n));
}
