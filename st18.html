                void main()
                {

                    for(i=0;str[i];i++)
                    {
                        if(str[i]>=0&&str[i]<=9)
                        {
                        strint[i]=(int)(str[i]-48);
                        }
                        else{
                           goto jump;
                        }
            
                    }
                    printf("\n");
                    for(i=0;str[i];i++)
                    {
                        printf(" %d ",(strint[i]));
                        //subrating 48 from the ascii value of the string will give us the

                    }
