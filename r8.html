#include<stdio.h>
                //find the arithmetic mean of an integer array using recursion
                int sum(int arr[], int n)
                {
                    if (n == 0 || n < 0) { //base case: when no elements are left in the array, return 0
                    return 0;
                    } else {
                        return (arr[n - 1] + sum(arr, n - 1)); //recursive call to add up all numbers in
                        //the array from last element to first
                        }
                        }
                        float find_mean(int arr[], int n){
                            int s = sum(arr, n);
                            return ((float)s / n); //return the arithmetic mean by dividing total sum by number of elements
                            }
                            int main(){
                                int arr[10];
                                printf("Enter 10 integers\n");
                                for (int i=0;i<10;i++){
                                    scanf("%d", &arr[i]);
                                    }
