7. CREATE TWO INTEGER STACKS IN A SINGLE ARRAY.

            //Implement two stacks in an array

#include<stdio.h>

int push(int ub,int *arr,int top,int val){
    if(top==ub)
    printf("Overflow condition\n");
    else{
        top++;
        arr[top]=val;
    }
    return top;
}

int pop(int lb,int *arr,int top){
    if(top<lb)
    printf("Underflow condition\n");
    else{
    printf("%d is popped\n",arr[top]);
    top--;
    }
    return top;
}
void print(int *arr,int lb,int top){
    int i;
    for(i=lb;i<=top;i++)
    printf("%d\t",arr[i]);
}


int main(){
    int arr[10],op,val,top=-1,ir=4;
    while(1){
        printf("Choose 1 option\n1.push in first stack\n2.Pop from first stack\n3.Push to another stack\n4.pop from another stack\n5.exit\n6.Print the first stack\n7.Print the second stack\n8.Print the full array\n");
        scanf("%d",&op);
        switch(op){
            case 1:
            printf("Enter the value to be pushed:");
            scanf("%d",&val);
            top=push(4,arr,top,val);
            break;
            case 2:
            top=pop(0,arr,top);
            break;
            case 3:
             printf("Enter the value to be pushed:");
            scanf("%d",&val);
            ir=push(9,arr,ir,val);
            break;
            case 4:
            ir=pop(5,arr,ir);
            break;
            case 5:
            exit(0);
            case 6:
            print(arr,0,top);
            break;
            case 7:
             print(arr,5,ir);
             break;
             case 8:
              print(arr,0,9);
              break;
              default:
              printf("Invalid choice\n");
              
        }}}
