#include<stdio.h>
            #include<string.h>
            #include<stdlib.h>
            
            struct Priorityqueue {
                char data[100]; 
                int prio;
                struct Priorityqueue *next;
            };
            
            typedef struct Priorityqueue pq;
            pq *front = NULL, *rear = NULL, *nn;
            
            void pqenqueue(char value[], int prio) {
                pq *sl, *search, *temp;
                nn = (pq *)malloc(sizeof(pq));
                strcpy(nn->data, value);
                nn->prio = prio;
                nn->next = NULL;
            
                temp = nn;
                if (front == NULL && rear == NULL) {
                    front = rear = nn;
                } else {
                    sl = NULL;
                    search = front;
                    while (search != NULL && temp->prio > search->prio) {
                        sl = search;
                        search = search->next;
                    }
                    if (sl != NULL) {
                        temp->next = sl->next;
                        sl->next = temp;
                    } else {
                        temp->next = front;
                        front = temp;
                    }
                }
            }
            
            void dequeue() {
                pq *temp;
                temp = front;
                if (rear == NULL && front == NULL) {
                    printf("\nEmpty");
                    return;
                } else {
                    front = front->next;
                    free(temp);
                }
            }
            
            void display() {
                pq *temp;
                temp = front;
                if (rear == NULL && front == NULL) {
                    printf("\nEmpty");
                    return;
                } else {
                    while (temp != rear) {
                        printf("%s (%d) -> ", temp->data, temp->prio);
                        temp = temp->next;
                    }
                    printf("%s (%d) -> ", temp->data, temp->prio);
                }
            }
            
            int main() {
                char value[100];
                int prio, ch;
                printf("\nPress 1: To insert value");
                printf("\nPress 2: To delete value");
                printf("\nPress 3: To display values");
                printf("\nPress 4: Exit the code");
                while (1) {
                    printf("\nEnter your choice: ");
                    scanf("%d", &ch);
            
                    switch (ch) {
                        case 1:
                            printf("\nEnter value to insert: ");
                            scanf("%s", value);
                            printf("\nEnter the priority: ");
                            scanf("%d", &prio);
                            pqenqueue(value, prio);
                            break;
            
                        case 2:
                            dequeue();
                            break;
            
                        case 3:
                            display();
                            break;
            
                        case 4:
                            printf("**Code has been terminated**\n");
                            exit(0);
                            break;
            
                        default:
                            printf("\nInvalid choice");
                            break;
                    }
                }
                return 0;
            }
