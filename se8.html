#include <stdio.h>
            #include <string.h>
            
            int binarySearchIterative(char arr[], char target, int length) {
                int left = 0;
                int right = length - 1;
            
                while (left <= right) {
                    int middle = left + (right - left) / 2;
            
                    if (arr[middle] == target) {
                        return middle; // Character found at this index
                    } else if (arr[middle] < target) {
                        left = middle + 1;
                    } else {
                        right = middle - 1;
                    }
                }
            
                return -1; // Character not found
            }
            
            int main() {
                char inputString[100];
                char searchChar;
            
                printf("Enter a sorted string: ");
                scanf("%s", inputString);
            
                printf("Enter a character to search: ");
                scanf(" %c", &searchChar); // Note the space before %c to consume any whitespace character
            
                int length = strlen(inputString);
                int result = binarySearchIterative(inputString, searchChar, length);
            
                if (result != -1) {
                    printf("Character '%c' found at position %d in the string.\n", searchChar, result);
                } else {
                    printf("Character '%c' not found in the string.\n", searchChar);
                }
            
                return 0;
            }
